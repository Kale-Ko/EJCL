012/**
013 * Contains methods useful for getting/setting nested values
014 *
015 * @version 4.0.0
016 * @since 1.0.0
017 */
018public class PathResolver {
019    protected static final Pattern arrayPathFixPattern = Pattern.compile("\\[([0-9]*)]");
020    protected static final Pattern pathSplitPattern = Pattern.compile("(?<!\\\\)(?:\\\\\\\\)*\\.");
021
022    private PathResolver() {
023    }
024
025    /**
026     * Resolve a value on an element
027     *
028     * @param element The element to resolve on
029     * @param path    The path to resolve to
030     *
031     * @return The value resolved
032     *
033     * @since 1.0.0
034     */
035    public static @Nullable Object resolve(@NotNull ParsedElement element, @NotNull String path) {
036        path = arrayPathFixPattern.matcher(path).replaceAll(".[$1]");
037
038        List<String> keys = Arrays.asList(pathSplitPattern.split(path));
039
040        ParsedElement resolved = element;
041
042        for (int i = 0; i < keys.size(); i++) {
043            if (resolved.isObject()) {
044                if (resolved.asObject().has(keys.get(i))) {
045                    resolved = resolved.asObject().get(keys.get(i));
046                } else {
047                    resolved = null;
048                    break;
049                }
050            } else if (resolved.isArray()) {
051                if (keys.get(i).startsWith("[") && keys.get(i).endsWith("]")) {
052                    int index = Integer.parseInt(arrayPathFixPattern.matcher(keys.get(i)).replaceAll(".[$1]"));
053
054                    if (index >= 0 && index < resolved.asArray().getSize()) {
055                        resolved = resolved.asArray().get(index);
056                    } else {
057                        resolved = null;
058                        break;
059                    }
060                } else {
061                    resolved = null;
062                    break;
063                }
064            } else {
065                resolved = null;
066                break;
067            }
068        }
069
070        if (resolved != null && resolved.isPrimitive()) {
071            return resolved.asPrimitive().get();
072        } else {
073            return null;
074        }
075    }

140    /**
141     * Update a value on an element
142     *
143     * @param element The element to update on
144     * @param path    The path to update
145     * @param value   The value to update to
146     * @param force   If the value should be force set (Create objects/arrays that don't exist)
147     *
148     * @return element for chaining
149     *
150     * @since 1.0.0
151     */
152    public static @NotNull ParsedElement update(@NotNull ParsedElement element, @NotNull String path, @Nullable Object value, boolean force) {
153        path = arrayPathFixPattern.matcher(path).replaceAll(".[$1]");
154
155        List<String> keys = Arrays.asList(pathSplitPattern.split(path));
156
157        ParsedElement resolved = element;
158
159        for (int i = 0; i < keys.size() - 1; i++) {
160            if (resolved.isObject()) {
161                if (resolved.asObject().has(keys.get(i))) {
162                    resolved = resolved.asObject().get(keys.get(i));
163                } else {
164                    if (force) {
165                        if (keys.get(i + 1).startsWith("[") && keys.get(i + 1).endsWith("]")) {
166                            resolved.asObject().set(keys.get(i), ParsedArray.create());
167                        } else if (keys.size() > i + 1) {
168                            resolved.asObject().set(keys.get(i), ParsedObject.create());
169                        } else {
170                            resolved.asObject().set(keys.get(i), ParsedPrimitive.fromNull());
171                        }
172
173                        resolved = resolved.asObject().get(keys.get(i));
174                    } else {
175                        resolved = null;
176                        break;
177                    }
178                }
179            } else if (resolved.isArray()) {
180                if (keys.get(i).startsWith("[") && keys.get(i).endsWith("]")) {
181                    int index = Integer.parseInt(arrayPathFixPattern.matcher(keys.get(i)).replaceAll(".[$1]"));
182
183                    if (index >= 0 && index < resolved.asArray().getSize()) {
184                        resolved = resolved.asArray().get(index);
185                    } else {
186                        if (force) {
187                            while (resolved.asArray().getSize() <= index) {
188                                if (keys.get(i + 1).startsWith("[") && keys.get(i + 1).endsWith("]")) {
189                                    resolved.asArray().add(ParsedArray.create());
190                                } else if (keys.size() > i + 1) {
191                                    resolved.asArray().add(ParsedObject.create());
192                                } else {
193                                    resolved.asArray().add(ParsedPrimitive.fromNull());
194                                }
195                            }
196
197                            resolved = resolved.asArray().get(index);
198                        } else {
199                            resolved = null;
200                            break;
201                        }
202                    }
203                } else {
204                    resolved = null;
205                    break;
206                }
207            } else {
208                resolved = null;
209                break;
210            }
211        }
212
213        String valueKey = keys.get(keys.size() - 1);
214
215        if (resolved != null) {
216            if (resolved.isObject()) {
217                if (resolved.asObject().has(valueKey)) {
218                    if (resolved.asObject().get(valueKey).isPrimitive()) {
219                        resolved.asObject().set(valueKey, ParsedPrimitive.from(value));
220                    }
221                } else {
222                    resolved.asObject().set(valueKey, ParsedPrimitive.from(value));
223                }
224            } else if (resolved.isArray() && (valueKey.startsWith("[") && valueKey.endsWith("]"))) {
225                int resolvedValueKey = Integer.parseInt(arrayPathFixPattern.matcher(valueKey).replaceAll(".[$1]"));
226
227                if (resolvedValueKey >= 0 && resolvedValueKey < resolved.asArray().getSize()) {
228                    if (resolved.asArray().get(resolvedValueKey).isPrimitive()) {
229                        resolved.asArray().set(resolvedValueKey, ParsedPrimitive.from(value));
230                    }
231                } else {
232                    resolved.asArray().add(ParsedPrimitive.from(value));
233                }
234            }
235        }
236
237        return element;
238    }

375    /**
376     * Get all the keys in an element
377     *
378     * @param element          The element to get the keys of
379     * @param path             The current path
380     * @param returnObjArrKeys Weather or not to return the keys for objects and arrays
381     *
382     * @return A list of all the keys
383     *
384     * @since 1.0.0
385     */
386    protected static @NotNull Set<String> getKeys(@NotNull ParsedElement element, @NotNull String path, boolean returnObjArrKeys) {
387        Set<String> keys = new HashSet<>();
388
389        if (element.isObject()) {
390            ParsedObject object = element.asObject();
391
392            for (Map.Entry<String, ParsedElement> entry : object.getEntries()) {
393                if (entry.getValue().isObject()) {
394                    if (returnObjArrKeys) {
395                        keys.add(path + entry.getKey().replace(".", "\\.")); // FIXME \. -> \\. -> split
396                    }
397                    keys.addAll(getKeys(entry.getValue(), path + entry.getKey().replace(".", "\\.") + ".", returnObjArrKeys));
398                } else if (entry.getValue().isArray()) {
399                    if (returnObjArrKeys) {
400                        keys.add(path + entry.getKey().replace(".", "\\."));
401                    }
402                    keys.addAll(getKeys(entry.getValue(), path + entry.getKey().replace(".", "\\."), returnObjArrKeys));
403                } else if (entry.getValue().isPrimitive()) {
404                    keys.add(path + entry.getKey().replace(".", "\\."));
405                }
406            }
407        } else if (element.isArray()) {
408            ParsedArray array = element.asArray();
409
410            for (int i = 0; i < array.getSize(); i++) {
411                if (array.get(i).isObject()) {
412                    if (returnObjArrKeys) {
413                        keys.add(path + "[" + i + "]");
414                    }
415                    keys.addAll(getKeys(array.get(i), path + "[" + i + "]" + ".", returnObjArrKeys));
416                } else if (array.get(i).isArray()) {
417                    if (returnObjArrKeys) {
418                        keys.add(path + "[" + i + "]");
419                    }
420                    keys.addAll(getKeys(array.get(i), path + "[" + i + "]", returnObjArrKeys));
421                } else if (array.get(i).isPrimitive()) {
422                    keys.add(path + "[" + i + "]");
423                }
424            }
425        }
426
427        return keys;
428    }

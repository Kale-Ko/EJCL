127    /**
128     * Get a path being stored
129     *
130     * @param path The path to get
131     *
132     * @return The value being stored
133     *
134     * @since 3.0.0
135     */
136    @Override
137    public @Nullable Object get(@NotNull String path) {
138        if (this.closed) {
139            throw new ConfigClosedException();
140        }
141
142        try {
143            while (!this.getConnected()) {
144                this.reconnectAttempts++;
145                if (this.reconnectAttempts > 5) {
146                    throw new MaximumReconnectsException();
147                }
148
149                this.connect();
150            }
151
152            assert this.connection != null;
153        } catch (IOException e) {
154            throw new MySQLException(e);
155        }
156
157        try (ResultSet result = MySQLHelper.query(this.connection, "SELECT type,value FROM " + this.table + " WHERE path=?", path)) {
158            Object parsedValue = null;
159
160            while (result.next()) {
161                String type = result.getString("type").toUpperCase();
162                ParsedPrimitive.PrimitiveType primitiveType = ParsedPrimitive.PrimitiveType.valueOf(type);
163                String value = result.getString("value");
164
165                switch (primitiveType) {
166                    case STRING: {
167                        parsedValue = value;
168                        break;
169                    }
170                    case BYTE: {
171                        parsedValue = Byte.parseByte(value);
172                        break;
173                    }
174                    case CHAR: {
175                        parsedValue = (char) Short.parseShort(value);
176                        break;
177                    }
178                    case SHORT: {
179                        parsedValue = Short.parseShort(value);
180                        break;
181                    }
182                    case INTEGER: {
183                        parsedValue = Integer.parseInt(value);
184                        break;
185                    }
186                    case LONG: {
187                        parsedValue = Long.parseLong(value);
188                        break;
189                    }
190                    case BIGINTEGER: {
191                        parsedValue = BigIntegerParser.parseWithFastParser(value);
192                        break;
193                    }
194                    case FLOAT: {
195                        parsedValue = Float.parseFloat(value);
196                        break;
197                    }
198                    case DOUBLE: {
199                        parsedValue = Double.parseDouble(value);
200                        break;
201                    }
202                    case BIGDECIMAL: {
203                        parsedValue = BigDecimalParser.parse(value);
204                        break;
205                    }
206                    case BOOLEAN: {
207                        parsedValue = Boolean.parseBoolean(value);
208                        break;
209                    }
210                    case NULL: {
211                        parsedValue = null;
212                        break;
213                    }
214                    default: {
215                        throw new RuntimeException();
216                    }
217                }
218            }
219
220            return parsedValue;
221        } catch (SQLException e) {
222            throw new MySQLException(e);
223        }
224    }

320    /**
321     * Set a path being stored
322     *
323     * @param path  The path to set
324     * @param value The value to set
325     *
326     * @since 3.0.0
327     */
328    @Override
329    public void set(@NotNull String path, @Nullable Object value) {
330        if (this.closed) {
331            throw new ConfigClosedException();
332        }
333
334        try {
335            while (!this.getConnected()) {
336                this.reconnectAttempts++;
337                if (this.reconnectAttempts > 5) {
338                    throw new MaximumReconnectsException();
339                }
340
341                this.connect();
342            }
343
344            assert this.connection != null;
345        } catch (IOException e) {
346            throw new MySQLException(e);
347        }
348
349        try {
350            if (value != null) {
351                ParsedPrimitive.PrimitiveType type = ParsedPrimitive.from(value).getType();
352                MySQLHelper.execute(this.connection, "REPLACE INTO " + this.table + " (path, type, value) VALUES (?, ?, ?);", path, type.name(), value.toString());
353            } else {
354                MySQLHelper.execute(this.connection, "DELETE FROM " + this.table + " WHERE path=?", path);
355            }
356        } catch (SQLException e) {
357            throw new MySQLException(e);
358        }
359    }

419    /**
420     * Connect to the server
421     *
422     * @throws IOException On connect error
423     * @since 3.0.0
424     */
425    public void connect() throws IOException {
426        synchronized (SAVELOAD_LOCK) {
427            try {
428                Properties properties = new Properties();
429                properties.put("characterEncoding", "utf8");
430                if (this.username != null) {
431                    properties.put("user", this.username);
432
433                    if (this.password != null) {
434                        properties.put("password", this.password);
435                    }
436                }
437
438                try {
439                    if (this.useMariadb) {
440                        Class.forName("org.mariadb.jdbc.Driver");
441                    } else {
442                        Class.forName("com.mysql.cj.jdbc.Driver");
443                    }
444                } catch (ClassNotFoundException e) {
445                    throw new DriverLoadException(e);
446                }
447
448                this.connection = DriverManager.getConnection("jdbc:" + (this.useMariadb ? "mariadb:" : "mysql:") + "//" + this.address.getHostString() + ":" + this.address.getPort() + "/" + this.database, properties);
449
450                if (this.connection.isValid(3)) {
451                    this.reconnectAttempts = 0;
452
453                    MySQLHelper.execute(this.connection, "CREATE TABLE IF NOT EXISTS " + this.table + " (path varchar(256) CHARACTER SET utf8 NOT NULL, type varchar(16) CHARACTER SET utf8 NOT NULL, value varchar(4096) CHARACTER SET utf8, PRIMARY KEY (path)) CHARACTER SET utf8;");
454
455                    try (ResultSet result = MySQLHelper.query(this.connection, "DESCRIBE " + this.table + ";")) {
456                        boolean typeExists = false;
457
458                        while (result.next()) {
459                            String name = result.getString("Field");
460                            if (name.equals("type")) {
461                                typeExists = true;
462                            }
463                        }
464
465                        if (!typeExists) {
466                            MySQLHelper.execute(this.connection, "ALTER TABLE " + this.table + " ADD COLUMN type varchar(16) CHARACTER SET utf8 NOT NULL AFTER path;");
467
468                            MySQLHelper.execute(this.connection, "UPDATE " + this.table + " SET type='STRING';");
469                        }
470                    } catch (SQLException e) {
471                        throw new IOException(e);
472                    }
473                } else {
474                    if (this.connection != null) {
475                        this.connection.close();
476                    }
477
478                    throw new IOException("Failed to connect: Connection is not valid");
479                }
480            } catch (SQLException e) {
481                throw new IOException(e);
482            }
483        }
484    }

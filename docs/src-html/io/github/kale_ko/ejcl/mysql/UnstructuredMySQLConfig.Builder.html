127    /**
128     * Get a path being stored
129     *
130     * @param path The path to get
131     *
132     * @return The value being stored
133     *
134     * @since 3.0.0
135     */
136    @Override
137    public @Nullable Object get(@NotNull String path) {
138        if (this.closed) {
139            throw new ConfigClosedException();
140        }
141
142        while (!this.getConnected()) {
143            this.reconnectAttempts++;
144
145            try {
146                this.connect();
147            } catch (IOException e) {
148                try {
149                    Thread.sleep((int) (Math.pow(2, this.reconnectAttempts) * 1000));
150                } catch (InterruptedException ex) {
151                    Thread.currentThread().interrupt();
152                }
153
154                if (this.reconnectAttempts > 5) {
155                    throw new MaximumReconnectsException(e);
156                }
157            }
158
159            if (this.reconnectAttempts > 5) {
160                throw new MaximumReconnectsException();
161            }
162        }
163
164        assert this.connection != null;
165
166        try (ResultSet result = MySQLHelper.query(this.connection, "SELECT type,value FROM " + this.table + " WHERE path=?", path)) {
167            Object parsedValue = null;
168
169            while (result.next()) {
170                String type = result.getString("type").toUpperCase();
171                ParsedPrimitive.PrimitiveType primitiveType = ParsedPrimitive.PrimitiveType.valueOf(type);
172                String value = result.getString("value");
173
174                switch (primitiveType) {
175                    case STRING: {
176                        parsedValue = value;
177                        break;
178                    }
179                    case BYTE: {
180                        parsedValue = Byte.parseByte(value);
181                        break;
182                    }
183                    case CHAR: {
184                        parsedValue = (char) Short.parseShort(value);
185                        break;
186                    }
187                    case SHORT: {
188                        parsedValue = Short.parseShort(value);
189                        break;
190                    }
191                    case INTEGER: {
192                        parsedValue = Integer.parseInt(value);
193                        break;
194                    }
195                    case LONG: {
196                        parsedValue = Long.parseLong(value);
197                        break;
198                    }
199                    case BIGINTEGER: {
200                        parsedValue = BigIntegerParser.parseWithFastParser(value);
201                        break;
202                    }
203                    case FLOAT: {
204                        parsedValue = Float.parseFloat(value);
205                        break;
206                    }
207                    case DOUBLE: {
208                        parsedValue = Double.parseDouble(value);
209                        break;
210                    }
211                    case BIGDECIMAL: {
212                        parsedValue = BigDecimalParser.parse(value);
213                        break;
214                    }
215                    case BOOLEAN: {
216                        parsedValue = Boolean.parseBoolean(value);
217                        break;
218                    }
219                    case NULL: {
220                        parsedValue = null;
221                        break;
222                    }
223                    default: {
224                        throw new RuntimeException();
225                    }
226                }
227            }
228
229            return parsedValue;
230        } catch (SQLException e) {
231            throw new MySQLException(e);
232        }
233    }

329    /**
330     * Set a path being stored
331     *
332     * @param path  The path to set
333     * @param value The value to set
334     *
335     * @since 3.0.0
336     */
337    @Override
338    public void set(@NotNull String path, @Nullable Object value) {
339        if (this.closed) {
340            throw new ConfigClosedException();
341        }
342
343        while (!this.getConnected()) {
344            this.reconnectAttempts++;
345
346            try {
347                this.connect();
348            } catch (IOException e) {
349                try {
350                    Thread.sleep((int) (Math.pow(2, this.reconnectAttempts) * 1000));
351                } catch (InterruptedException ex) {
352                    Thread.currentThread().interrupt();
353                }
354
355                if (this.reconnectAttempts > 5) {
356                    throw new MaximumReconnectsException(e);
357                }
358            }
359
360            if (this.reconnectAttempts > 5) {
361                throw new MaximumReconnectsException();
362            }
363        }
364
365        assert this.connection != null;
366
367        try {
368            if (value != null) {
369                ParsedPrimitive.PrimitiveType type = ParsedPrimitive.from(value).getType();
370                MySQLHelper.execute(this.connection, "REPLACE INTO " + this.table + " (path, type, value) VALUES (?, ?, ?);", path, type.name(), value.toString());
371            } else {
372                MySQLHelper.execute(this.connection, "DELETE FROM " + this.table + " WHERE path=?", path);
373            }
374        } catch (SQLException e) {
375            throw new MySQLException(e);
376        }
377    }

437    /**
438     * Connect to the server
439     *
440     * @throws IOException On connect error
441     * @since 3.0.0
442     */
443    public void connect() throws IOException {
444        synchronized (SAVELOAD_LOCK) {
445            try {
446                Properties properties = new Properties();
447                properties.put("characterEncoding", "utf8");
448                if (this.username != null) {
449                    properties.put("user", this.username);
450
451                    if (this.password != null) {
452                        properties.put("password", this.password);
453                    }
454                }
455
456                try {
457                    if (this.useMariadb) {
458                        Class.forName("org.mariadb.jdbc.Driver");
459                    } else {
460                        Class.forName("com.mysql.cj.jdbc.Driver");
461                    }
462                } catch (ClassNotFoundException e) {
463                    throw new DriverLoadException(e);
464                }
465
466                this.connection = DriverManager.getConnection("jdbc:" + (this.useMariadb ? "mariadb:" : "mysql:") + "//" + this.address.getHostString() + ":" + this.address.getPort() + "/" + this.database, properties);
467
468                if (this.connection.isValid(3)) {
469                    this.reconnectAttempts = 0;
470
471                    MySQLHelper.execute(this.connection, "CREATE TABLE IF NOT EXISTS " + this.table + " (path varchar(256) CHARACTER SET utf8 NOT NULL, type varchar(16) CHARACTER SET utf8 NOT NULL, value varchar(4096) CHARACTER SET utf8, PRIMARY KEY (path)) CHARACTER SET utf8;");
472
473                    try (ResultSet result = MySQLHelper.query(this.connection, "DESCRIBE " + this.table + ";")) {
474                        boolean typeExists = false;
475
476                        while (result.next()) {
477                            String name = result.getString("Field");
478                            if (name.equals("type")) {
479                                typeExists = true;
480                            }
481                        }
482
483                        if (!typeExists) {
484                            MySQLHelper.execute(this.connection, "ALTER TABLE " + this.table + " ADD COLUMN type varchar(16) CHARACTER SET utf8 NOT NULL AFTER path;");
485
486                            MySQLHelper.execute(this.connection, "UPDATE " + this.table + " SET type='STRING';");
487                        }
488                    } catch (SQLException e) {
489                        throw new IOException(e);
490                    }
491                } else {
492                    if (this.connection != null) {
493                        this.connection.close();
494                    }
495
496                    throw new IOException("Failed to connect: Connection is not valid");
497                }
498            } catch (SQLException e) {
499                throw new IOException(e);
500            }
501        }
502    }

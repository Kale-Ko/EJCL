209    /**
210     * Connect to the server
211     *
212     * @throws IOException On connect error
213     * @since 1.0.0
214     */
215    public void connect() throws IOException {
216        synchronized (SAVELOAD_LOCK) {
217            try {
218                Properties properties = new Properties();
219                properties.put("characterEncoding", "utf8");
220                if (this.username != null) {
221                    properties.put("user", this.username);
222
223                    if (this.password != null) {
224                        properties.put("password", this.password);
225                    }
226                }
227
228                try {
229                    if (this.useMariadb) {
230                        Class.forName("org.mariadb.jdbc.Driver");
231                    } else {
232                        Class.forName("com.mysql.cj.jdbc.Driver");
233                    }
234                } catch (ClassNotFoundException e) {
235                    throw new DriverLoadException(e);
236                }
237
238                this.connection = DriverManager.getConnection("jdbc:" + (this.useMariadb ? "mariadb:" : "mysql:") + "//" + this.address.getHostString() + ":" + this.address.getPort() + "/" + this.database, properties);
239
240                if (this.connection.isValid(3)) {
241                    this.reconnectAttempts = 0;
242
243                    MySQLHelper.execute(this.connection, "CREATE TABLE IF NOT EXISTS " + this.table + " (path varchar(256) CHARACTER SET utf8 NOT NULL, type varchar(16) CHARACTER SET utf8 NOT NULL, value varchar(4096) CHARACTER SET utf8, PRIMARY KEY (path)) CHARACTER SET utf8;");
244
245                    try (ResultSet result = MySQLHelper.query(this.connection, "DESCRIBE " + this.table + ";")) {
246                        boolean typeExists = false;
247
248                        while (result.next()) {
249                            String name = result.getString("Field");
250                            if (name.equals("type")) {
251                                typeExists = true;
252                            }
253                        }
254
255                        if (!typeExists) {
256                            MySQLHelper.execute(this.connection, "ALTER TABLE " + this.table + " ADD COLUMN type varchar(16) CHARACTER SET utf8 NOT NULL AFTER path;");
257
258                            MySQLHelper.execute(this.connection, "UPDATE " + this.table + " SET type='STRING';");
259                        }
260                    } catch (SQLException e) {
261                        throw new IOException(e);
262                    }
263                } else {
264                    if (this.connection != null) {
265                        this.connection.close();
266                    }
267
268                    throw new IOException("Failed to connect: Connection is not valid");
269                }
270            } catch (SQLException e) {
271                throw new IOException(e);
272            }
273        }
274    }

276    /**
277     * Load the config
278     *
279     * @param save Weather to save the config after loaded (To update the template)
280     *
281     * @throws IOException On load error
282     * @since 1.3.0
283     */
284    @Override
285    public void load(boolean save) throws IOException {
286        if (this.closed) {
287            throw new ConfigClosedException();
288        }
289
290        while (!this.getConnected()) {
291            this.reconnectAttempts++;
292
293            try {
294                this.connect();
295            } catch (IOException e) {
296                try {
297                    Thread.sleep((int) (Math.pow(2, this.reconnectAttempts) * 1000));
298                } catch (InterruptedException ex) {
299                    Thread.currentThread().interrupt();
300                }
301
302                if (this.reconnectAttempts > 5) {
303                    throw new MaximumReconnectsException(e);
304                }
305            }
306
307            if (this.reconnectAttempts > 5) {
308                throw new MaximumReconnectsException();
309            }
310        }
311        assert this.connection != null;
312
313        synchronized (SAVELOAD_LOCK) {
314            ParsedObject object = ParsedObject.create();
315
316            try (ResultSet result = MySQLHelper.query(this.connection, "SELECT path,type,value FROM " + this.table)) {
317                while (result.next()) {
318                    String path = result.getString("path");
319                    String type = result.getString("type").toUpperCase();
320                    ParsedPrimitive.PrimitiveType primitiveType = ParsedPrimitive.PrimitiveType.valueOf(type);
321                    String value = result.getString("value");
322
323                    ParsedPrimitive element;
324                    switch (primitiveType) {
325                        case STRING: {
326                            element = ParsedPrimitive.fromString(value);
327                            break;
328                        }
329                        case BYTE: {
330                            element = ParsedPrimitive.fromByte(Byte.parseByte(value));
331                            break;
332                        }
333                        case CHAR: {
334                            element = ParsedPrimitive.fromChar((char) Short.parseShort(value));
335                            break;
336                        }
337                        case SHORT: {
338                            element = ParsedPrimitive.fromShort(Short.parseShort(value));
339                            break;
340                        }
341                        case INTEGER: {
342                            element = ParsedPrimitive.fromInteger(Integer.parseInt(value));
343                            break;
344                        }
345                        case LONG: {
346                            element = ParsedPrimitive.fromLong(Long.parseLong(value));
347                            break;
348                        }
349                        case BIGINTEGER: {
350                            element = ParsedPrimitive.fromBigInteger(BigIntegerParser.parseWithFastParser(value));
351                            break;
352                        }
353                        case FLOAT: {
354                            element = ParsedPrimitive.fromFloat(Float.parseFloat(value));
355                            break;
356                        }
357                        case DOUBLE: {
358                            element = ParsedPrimitive.fromDouble(Double.parseDouble(value));
359                            break;
360                        }
361                        case BIGDECIMAL: {
362                            element = ParsedPrimitive.fromBigDecimal(BigDecimalParser.parse(value));
363                            break;
364                        }
365                        case BOOLEAN: {
366                            element = ParsedPrimitive.fromBoolean(Boolean.parseBoolean(value));
367                            break;
368                        }
369                        case NULL: {
370                            element = ParsedPrimitive.fromNull();
371                            break;
372                        }
373                        default: {
374                            throw new RuntimeException();
375                        }
376                    }
377
378                    PathResolver.updateElement(object, path, element, true);
379                }
380            } catch (SQLException e) {
381                throw new IOException(e);
382            }
383
384            this.config = this.processor.toObject(object, this.clazz);
385            this.configExpires = Instant.now().plus(this.cacheLength);
386
387            this.configBackup = object;
388        }
389
390        if (save) {
391            this.save();
392        }
393    }

395    /**
396     * Save the config to the server
397     *
398     * @throws IOException On save error
399     * @since 1.0.0
400     */
401    @Override
402    public void save() throws IOException {
403        if (this.closed) {
404            throw new ConfigClosedException();
405        }
406        if (this.config == null || this.configBackup == null) {
407            throw new ConfigNotLoadedException();
408        }
409
410        while (!this.getConnected()) {
411            this.reconnectAttempts++;
412
413            try {
414                this.connect();
415            } catch (IOException e) {
416                try {
417                    Thread.sleep((int) (Math.pow(2, this.reconnectAttempts) * 1000));
418                } catch (InterruptedException ex) {
419                    Thread.currentThread().interrupt();
420                }
421
422                if (this.reconnectAttempts > 5) {
423                    throw new MaximumReconnectsException(e);
424                }
425            }
426
427            if (this.reconnectAttempts > 5) {
428                throw new MaximumReconnectsException();
429            }
430        }
431        assert this.connection != null;
432
433        synchronized (SAVELOAD_LOCK) {
434            ParsedObject object = this.processor.toElement(this.config).asObject();
435            Set<String> keys = PathResolver.getKeys(object, false);
436            Set<String> oldKeys = PathResolver.getKeys(this.configBackup, false);
437
438            Set<String> toDelete = new HashSet<>(oldKeys);
439            toDelete.removeAll(keys);
440
441            List<String> queryArgs = new ArrayList<>();
442            for (String key : keys) {
443                ParsedElement valueElement = PathResolver.resolveElement(object, key);
444                ParsedElement oldValueElement = PathResolver.resolveElement(this.configBackup, key);
445
446                if (valueElement != null && oldValueElement != null) {
447                    ParsedPrimitive value = valueElement.asPrimitive();
448                    ParsedPrimitive oldValue = oldValueElement.asPrimitive();
449
450                    Object valueObj = value.get();
451                    Object oldValueObj = oldValue.get();
452
453                    if (!((value.isNull() && oldValue.isNull()) || (!value.isNull() && (valueObj == oldValueObj || valueObj.equals(oldValueObj))))) {
454                        queryArgs.add(key);
455                        queryArgs.add(value.getType().name());
456                        queryArgs.add(valueObj != null ? valueObj.toString() : "null");
457                    }
458                }
459            }
460
461            this.configBackup = object;
462
463            try {
464                if (!queryArgs.isEmpty()) {
465                    MySQLHelper.executeBatch(this.connection, "REPLACE INTO " + this.table + " (path, type, value) VALUES (?, ?, ?);", 3, queryArgs);
466                }
467
468                if (!toDelete.isEmpty()) {
469                    MySQLHelper.executeBatch(this.connection, "DELETE FROM " + this.table + " WHERE path=?;", 1, List.copyOf(toDelete));
470                }
471            } catch (SQLException e) {
472                throw new IOException(e);
473            }
474        }
475    }

477    /**
478     * Close the config
479     *
480     * @throws IOException On close error
481     * @since 1.0.0
482     */
483    @Override
484    public void close() throws IOException {
485        if (this.closed) {
486            throw new ConfigClosedException();
487        }
488
489        this.closed = true;
490
491        try {
492            if (this.connection != null) {
493                this.connection.close();
494            }
495        } catch (SQLException e) {
496            throw new IOException(e);
497        }
498    }

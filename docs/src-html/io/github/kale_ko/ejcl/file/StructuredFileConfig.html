001package io.github.kale_ko.ejcl.file;
002
003import io.github.kale_ko.ejcl.StructuredConfig;
004import io.github.kale_ko.ejcl.exception.ConfigClosedException;
005import io.github.kale_ko.ejcl.exception.ConfigNotLoadedException;
006import java.io.IOException;
007import java.nio.file.Files;
008import java.nio.file.Path;
009import java.nio.file.StandardOpenOption;
010import org.jetbrains.annotations.NotNull;
011
012/**
013 * A Structured File Config for storing data in a file
014 *
015 * @param <T> The type of the data being stored
016 *
017 * @version 5.0.0
018 * @since 1.0.0
019 */
020public abstract class StructuredFileConfig<T> extends StructuredConfig<T> {
021    /**
022     * The file where data is being stored
023     *
024     * @since 1.0.0
025     */
026    protected final @NotNull Path file;
027
028    /**
029     * The lock used when saving and loading the config
030     *
031     * @since 3.8.0
032     */
033    protected final @NotNull Object SAVELOAD_LOCK = new Object();
034
035    /**
036     * If this config is closed
037     *
038     * @since 1.0.0
039     */
040    protected boolean closed = false;
041
042    /**
043     * Create a new FileConfig
044     *
045     * @param clazz The class of the data being stored
046     * @param file  The file where data is being stored
047     *
048     * @since 1.0.0
049     */
050    protected StructuredFileConfig(@NotNull Class<T> clazz, @NotNull Path file) {
051        this(clazz, file, false);
052    }
053
054    /**
055     * Create a new FileConfig
056     *
057     * @param clazz          The class of the data being stored
058     * @param file           The file where data is being stored
059     * @param supportsArrays If the config supports storing arrays
060     *
061     * @since 5.0.0
062     */
063    protected StructuredFileConfig(@NotNull Class<T> clazz, @NotNull Path file, boolean supportsArrays) {
064        super(clazz, supportsArrays);
065
066        this.file = file;
067    }
068
069    /**
070     * Get the file where data is being stored
071     *
072     * @return The file where data is being stored
073     *
074     * @since 1.0.0
075     */
076    public @NotNull Path getFile() {
077        return this.file;
078    }
079
080    /**
081     * Get if the config is loaded
082     *
083     * @return If the config is loaded
084     *
085     * @since 1.0.0
086     */
087    @Override
088    public boolean getLoaded() {
089        return this.config != null;
090    }
091
092    /**
093     * Create a blank config file
094     *
095     * @return The config bytes
096     *
097     * @throws IOException On create error
098     * @since 1.0.0
099     */
100    protected abstract byte @NotNull [] create() throws IOException;
101
102    /**
103     * Load the config
104     *
105     * @param save Weather to save the config after loaded (To update the template)
106     *
107     * @throws IOException On load error
108     * @since 1.3.0
109     */
110    @Override
111    public abstract void load(boolean save) throws IOException;
112
113    /**
114     * Load the config from file
115     *
116     * @return The file bytes
117     *
118     * @throws IOException On load error
119     * @since 1.0.0
120     */
121    protected byte @NotNull [] loadRaw() throws IOException {
122        if (!Files.exists(this.file)) {
123            return this.create();
124        }
125
126        return Files.readAllBytes(this.file);
127    }
128
129    /**
130     * Save the config to file
131     *
132     * @throws IOException On save error
133     * @since 1.0.0
134     */
135    @Override
136    public void save() throws IOException {
137        if (this.closed) {
138            throw new ConfigClosedException();
139        }
140        if (this.config == null) {
141            throw new ConfigNotLoadedException();
142        }
143
144        synchronized (SAVELOAD_LOCK) {
145            if (!Files.exists(this.file)) {
146                Files.createFile(this.file);
147            }
148            Files.write(this.file, this.saveRaw(), StandardOpenOption.TRUNCATE_EXISTING);
149        }
150    }
151
152    /**
153     * Save the config to bytes
154     *
155     * @return The config bytes
156     *
157     * @throws IOException On save error
158     * @since 1.0.0
159     */
160    protected abstract byte @NotNull [] saveRaw() throws IOException;
161
162    /**
163     * Close the config
164     *
165     * @throws IOException On close error
166     * @since 1.0.0
167     */
168    @Override
169    public void close() throws IOException {
170        if (this.closed) {
171            throw new ConfigClosedException();
172        }
173
174        this.closed = true;
175    }
176
177    /**
178     * Get if the config is closed
179     *
180     * @return If the config is closed
181     *
182     * @since 1.0.0
183     */
184    public boolean isClosed() {
185        return this.closed;
186    }
187}

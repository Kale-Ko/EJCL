020/**

032    /**
033     * Registers the legacy default type processors for a given builder. This also unregisters all other default type processors.
034     *
035     * @param builder The builder to register them to
036     *
037     * @since 5.0.0
038     * @deprecated Only for use in migration.
039     */
040    @Deprecated(forRemoval=true)
041    public static void register(@NotNull ObjectProcessor.Builder builder) {
042        if (builder.hasTypeProcessor(UUID.class)) {
043            builder.removeTypeProcessor(UUID.class);
044        }
045        builder.createTypeProcessor(UUID.class, new TypeProcessor() {
046            @Override
047            public @NotNull ParsedElement toElement(@Nullable Object object) {
048                if (object == null) {
049                    return ParsedPrimitive.fromNull();
050                }
051
052                if (object instanceof UUID) {
053                    return ParsedPrimitive.fromString(object.toString());
054                } else {
055                    throw new InvalidParameterException("object must be UUID");
056                }
057            }
058
059            @Override
060            public @Nullable Object toObject(@NotNull ParsedElement element) {
061                if (element.isPrimitive() && element.asPrimitive().isNull()) {
062                    return null;
063                }
064
065                if (element.isPrimitive() && element.asPrimitive().isString()) {
066                    return UUID.fromString(element.asPrimitive().asString());
067                } else {
068                    throw new InvalidParameterException("object must be String");
069                }
070            }
071        });
072
073        if (builder.hasTypeProcessor(URI.class)) {
074            builder.removeTypeProcessor(URI.class);
075        }
076        builder.createTypeProcessor(URI.class, new TypeProcessor() {
077            @Override
078            public @NotNull ParsedElement toElement(@Nullable Object object) {
079                if (object == null) {
080                    return ParsedPrimitive.fromNull();
081                }
082

086                    throw new InvalidParameterException("object must be URI");

089
090            @Override
091            public @Nullable Object toObject(@NotNull ParsedElement element) {
092                if (element.isPrimitive() && element.asPrimitive().isNull()) {
093                    return null;
094                }
095
096                if (element.isPrimitive() && element.asPrimitive().isString()) {
097                    try {
098                        return new URI(element.asPrimitive().asString());
099                    } catch (URISyntaxException e) {
100                        throw new InvalidParameterException("object is an invalid URI");
101                    }
102                } else {
103                    throw new InvalidParameterException("object must be String");
104                }
105            }
106        });
107
108        if (builder.hasTypeProcessor(URL.class)) {
109            builder.removeTypeProcessor(URL.class);

111        builder.createTypeProcessor(URL.class, new TypeProcessor() {
112            @Override
113            public @NotNull ParsedElement toElement(@Nullable Object object) {
114                if (object == null) {
115                    return ParsedPrimitive.fromNull();
116                }
117
118                if (object instanceof URL) {
119                    return ParsedPrimitive.fromString(object.toString());
120                } else {
121                    throw new InvalidParameterException("object must be URL");
122                }
123            }
124
125            @Override
126            public @Nullable Object toObject(@NotNull ParsedElement element) {
127                if (element.isPrimitive() && element.asPrimitive().isNull()) {
128                    return null;

153                if (object instanceof Path) {
154                    return ParsedPrimitive.fromString(((Path) object).toString());

156                    throw new InvalidParameterException("object must be Path");

205        if (builder.hasTypeProcessor(InetAddress.class)) {
206            builder.removeTypeProcessor(InetAddress.class);
207        }
208        builder.createTypeProcessor(InetAddress.class, new TypeProcessor() {
209            @Override
210            public @NotNull ParsedElement toElement(@Nullable Object object) {
211                if (object == null) {
212                    return ParsedPrimitive.fromNull();
213                }
214
215                if (object instanceof InetAddress) {
216                    byte[] address = ((InetAddress) object).getAddress();
217
218                    if (address.length == 4) {
219                        String[] addressBytes = new String[4];
220
221                        for (int i = 0; i < 4; i++) {
222                            addressBytes[i] = Integer.toUnsignedString(address[i] & 0xff, 10);
223                        }
224
225                        return ParsedPrimitive.fromString(String.join(".", addressBytes));
226                    } else if (address.length == 16) {
227                        String[] addressBytes = new String[8];
228
229                        for (int i = 0; i < 8; i++) {
230                            addressBytes[i] = Integer.toUnsignedString(((address[i * 2] & 0xff) << 8) + (address[(i * 2) + 1] & 0xff), 16);
231                        }
232
233                        return ParsedPrimitive.fromString(String.join(":", addressBytes));
234                    } else {
235                        throw new InvalidParameterException("InetAddress must be IPv4 or IPv6");
236                    }
237                } else {
238                    throw new InvalidParameterException("object must be InetAddress");
239                }
240            }
241
242            @Override
243            public @Nullable Object toObject(@NotNull ParsedElement element) {
244                if (element.isPrimitive() && element.asPrimitive().isNull()) {
245                    return null;
246                }
247
248                if (element.isPrimitive() && element.asPrimitive().isString()) {
249                    try {
250                        String[] addressBytes = element.asPrimitive().asString().split("[.:]");
251
252                        if (addressBytes.length == 4) {
253                            byte[] address = new byte[4];
254
255                            for (int i = 0; i < 4; i++) {
256                                address[i] = (byte) (Integer.parseUnsignedInt(addressBytes[i], 10) & 0xff);
257                            }
258
259                            return InetAddress.getByAddress(address);
260                        } else if (addressBytes.length == 8) {
261                            byte[] address = new byte[16];
262
263                            for (int i = 0; i < 8; i++) {
264                                address[i * 2] = (byte) (Integer.parseUnsignedInt(addressBytes[i], 16) & 0xff);
265                                address[(i * 2) + 1] = (byte) (Integer.parseUnsignedInt(addressBytes[i], 16) & 0xff);
266                            }
267
268                            return InetAddress.getByAddress(address);
269                        } else {
270                            throw new InvalidParameterException("InetAddress must be IPv4 or IPv6");
271                        }
272                    } catch (UnknownHostException e) {
273                        throw new InvalidParameterException("object is an invalid InetAddress");
274                    }
275                } else {
276                    throw new InvalidParameterException("object must be String");
277                }
278            }
279        });
280
281        if (builder.hasTypeProcessor(InetSocketAddress.class)) {
282            builder.removeTypeProcessor(InetSocketAddress.class);
283        }
284        builder.createTypeProcessor(InetSocketAddress.class, new TypeProcessor() {
285            @Override
286            public @NotNull ParsedElement toElement(@Nullable Object object) {
287                if (object == null) {
288                    return ParsedPrimitive.fromNull();
289                }
290
291                if (object instanceof InetSocketAddress) {
292                    byte[] address = ((InetSocketAddress) object).getAddress().getAddress();
293                    int port = ((InetSocketAddress) object).getPort();
294
295                    if (address.length == 4) {
296                        String[] addressBytes = new String[4];
297
298                        for (int i = 0; i < 4; i++) {
299                            addressBytes[i] = Integer.toUnsignedString(address[i] & 0xff, 10);
300                        }
301
302                        return ParsedPrimitive.fromString(String.join(".", addressBytes) + ":" + Integer.toUnsignedString(port, 10));
303                    } else if (address.length == 16) {
304                        String[] addressBytes = new String[8];
305
306                        for (int i = 0; i < 8; i++) {
307                            addressBytes[i] = Integer.toUnsignedString(((address[i * 2] & 0xff) << 8) + (address[(i * 2) + 1] & 0xff), 16);
308                        }
309
310                        return ParsedPrimitive.fromString("[" + String.join(":", addressBytes) + "]:" + Integer.toUnsignedString(port, 10));
311                    } else {
312                        throw new InvalidParameterException("InetSocketAddress must be IPv4 or IPv6");
313                    }
314                } else {
315                    throw new InvalidParameterException("object must be InetSocketAddress");
316                }
317            }
318
319            @Override
320            public @Nullable Object toObject(@NotNull ParsedElement element) {
321                if (element.isPrimitive() && element.asPrimitive().isNull()) {
322                    return null;
323                }
324
325                if (element.isPrimitive() && element.asPrimitive().isString()) {
326                    try {
327                        String[] addressBytes = element.asPrimitive().asString().replace("[", "").replace("]", "").split("[.:]");
328
329                        if (addressBytes.length == 5) {
330                            byte[] address = new byte[4];
331                            int port;
332
333                            for (int i = 0; i < 4; i++) {
334                                address[i] = (byte) (Integer.parseUnsignedInt(addressBytes[i], 10) & 0xff);
335                            }
336
337                            port = Integer.parseUnsignedInt(addressBytes[4], 10);
338
339                            return new InetSocketAddress(InetAddress.getByAddress(address), port);
340                        } else if (addressBytes.length == 9) {
341                            byte[] address = new byte[16];
342                            int port;
343
344                            for (int i = 0; i < 8; i++) {
345                                address[i * 2] = (byte) (Integer.parseUnsignedInt(addressBytes[i], 16) & 0xff);
346                                address[(i * 2) + 1] = (byte) (Integer.parseUnsignedInt(addressBytes[i], 16) & 0xff);
347                            }
348
349                            port = Integer.parseUnsignedInt(addressBytes[8], 10);
350
351                            return new InetSocketAddress(InetAddress.getByAddress(address), port);
352                        } else {
353                            throw new InvalidParameterException("InetSocketAddress must be IPv4 or IPv6");
354                        }
355                    } catch (UnknownHostException e) {
356                        throw new InvalidParameterException("object is an invalid InetSocketAddress");
357                    }
358                } else {
359                    throw new InvalidParameterException("object must be String");
360                }
361            }
362        });
363

371                    return ParsedPrimitive.fromNull();

376                } else {
377                    throw new InvalidParameterException("object must be Calendar");
378                }
379            }
380
381            @Override
382            public @Nullable Object toObject(@NotNull ParsedElement element) {
383                if (element.isPrimitive() && element.asPrimitive().isNull()) {
384                    return null;
385                }
386
387                if (element.isPrimitive() && element.asPrimitive().isString()) {
388                    try {
389                        Calendar calendar = Calendar.getInstance();
390                        calendar.setTime(DateFormat.getDateInstance(DateFormat.DEFAULT).parse(element.asPrimitive().asString()));
391                        return calendar;
392                    } catch (ParseException e) {
393                        throw new InvalidParameterException("object is an invalid Date");
394                    }
395                } else {
396                    throw new InvalidParameterException("object must be String");
397                }
398            }
399        });

414                    throw new InvalidParameterException("object must be Date");
415                }
416            }
417
418            @Override
419            public @Nullable Object toObject(@NotNull ParsedElement element) {
420                if (element.isPrimitive() && element.asPrimitive().isNull()) {
421                    return null;
422                }
423
424                if (element.isPrimitive() && element.asPrimitive().isString()) {
425                    try {
426                        return DateFormat.getDateInstance(DateFormat.DEFAULT).parse(element.asPrimitive().asString());
427                    } catch (ParseException e) {
428                        throw new InvalidParameterException("object is an invalid Date");
429                    }
430                } else {
431                    throw new InvalidParameterException("object must be String");
432                }
433            }
434        });

436        if (builder.hasTypeProcessor(Instant.class)) {
437            builder.removeTypeProcessor(Instant.class);
438        }
439        builder.createTypeProcessor(Instant.class, new TypeProcessor() {
440            @Override
441            public @NotNull ParsedElement toElement(@Nullable Object object) {
442                if (object == null) {
443                    return ParsedPrimitive.fromNull();
444                }
445
446                if (object instanceof Instant) {
447                    return ParsedPrimitive.fromLong(((Instant) object).toEpochMilli());
448                } else {
449                    throw new InvalidParameterException("object must be Instant");
450                }
451            }
452
453            @Override
454            public @Nullable Object toObject(@NotNull ParsedElement element) {
455                if (element.isPrimitive() && element.asPrimitive().isNull()) {
456                    return null;
457                }
458
459                if (element.isPrimitive() && element.asPrimitive().isLong()) {
460                    return Instant.ofEpochMilli(element.asPrimitive().asLong());
461                } else if (element.isPrimitive() && element.asPrimitive().isString()) {
462                    return Instant.ofEpochMilli(Long.parseLong(element.asPrimitive().asString()));
463                } else {
464                    throw new InvalidParameterException("object must be Long");
465                }
466            }
467        });
468    }
